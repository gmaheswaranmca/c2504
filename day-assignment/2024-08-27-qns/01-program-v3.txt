 public class Service
 {
     public static long FindSumOfThirdAndFourth(long[] arr)
     {
         if (arr.Length < 4)
         {
             Console.WriteLine("3rd or 4th element is not there");
             throw new IndexOutOfRangeException();
         }
         return arr[2] + arr[3];
     }
 }
